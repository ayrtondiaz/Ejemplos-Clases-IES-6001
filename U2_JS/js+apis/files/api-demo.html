Consigna 4: Fetch básico con async/await

api.js — Función principal 
// (a) función async que haga fetch()
const cargarPersonajes = async () => {

  // (d) estado "Cargando..."
  container.innerHTML = '<p class="loading">Cargando...</p>';

  // (c) try/catch para manejar errores
  try {
    const response = await fetch(
      "https://rickandmortyapi.com/api/character"
    );

    // (b) verificar response.ok
    if (!response.ok) {
      throw new Error(`Error HTTP: ${response.status}`);
    }

    const data = await response.json();

    // (e) renderizar como tarjetas con .map()
    container.innerHTML = data.results.map(p => `
      <article class="card">
        <img src="${p.image}" alt="${p.name}">
        <h3>${p.name}</h3>
        <p>${p.species} - ${p.status}</p>
      </article>
    `).join("");

  } catch (error) {
    // (f) mostrar mensaje de error visible
    container.innerHTML =
      `<p class="error">❌ ${error.message}</p>`;
  }
};

cargarPersonajes(); // ejecutar al cargar la página

Consigna 5: Buscador dinámico con addEventListener("input")

api.js — Buscador 
let todosLosPersonajes = []; // cache global

// Cargar datos y guardar en cache
const cargarYGuardar = async () => {
  try {
    const res = await fetch("https://rickandmortyapi.com/api/character");
    if (!res.ok) throw new Error(`Error: ${res.status}`);
    const data = await res.json();
    todosLosPersonajes = data.results;
    renderizar(todosLosPersonajes);
  } catch (err) {
    mostrarError(err.message);
  }
};

// (a) Escuchar cambios en el input
document.querySelector("#busqueda")
  .addEventListener("input", (e) => {
    const texto = e.target.value.trim().toLowerCase();

    // (a) no buscar si hay menos de 3 caracteres
    if (texto.length > 0 && texto.length < 3) {
      mostrarHint("Escribí al menos 3 caracteres");
      return;
    }

    // Filtrar localmente con .filter()
    const filtrados = todosLosPersonajes.filter(p =>
      p.name.toLowerCase().includes(texto)
    );

    // (d) mostrar "No se encontraron resultados"
    if (filtrados.length === 0) {
      mostrarVacio();
      return;
    }

    // (c) actualizar las tarjetas
    renderizar(filtrados);
  });
