productos.js — Lógica principal 
// Array de productos
const productos = [
  { id: 1, nombre: "Laptop", precio: 1200, categoria: "Electrónica", enStock: true },
  { id: 2, nombre: "Mouse",  precio: 25,   categoria: "Electrónica", enStock: true },
  // ... 8 productos
];

// Función de filtrado — encadena .filter() por cada criterio
const filtrarProductos = () => {
  const texto = busqueda.value.toLowerCase();
  const cat = categoria.value;
  const precioMax = Number(precio.value);
  const soloStock = stock.checked;

  const resultado = productos
    .filter(p => p.nombre.toLowerCase().includes(texto))  // búsqueda
    .filter(p => cat === "todas" || p.categoria === cat)   // categoría
    .filter(p => p.precio <= precioMax)                    // precio
    .filter(p => !soloStock || p.enStock);                 // stock

  renderProductos(resultado);
};

// Renderizar con .map() + innerHTML
const renderProductos = (lista) => {
  container.innerHTML = lista.map(p => `
    <article class="card ${!p.enStock ? 'sin-stock' : ''}">
      <h3>${p.nombre}</h3>
      <p class="precio">$${p.precio}</p>
      <span class="cat">${p.categoria}</span>
    </article>
  `).join("");
};

// Escuchar TODOS los filtros
[busqueda, categoria, precio, stock].forEach(el =>
  el.addEventListener("input", filtrarProductos)
);
